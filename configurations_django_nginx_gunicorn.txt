Nginx Configuration:


server {
    listen 80;
    server_name gestion_notes.com;  

    location / {
        proxy_pass http://127.0.0.1:8000;  # Proxy vers Gunicorn
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }

    location /static/ {
        alias /path/to/note/project/static/; 
    }

    location /media/ {
        alias /path/to/gestion_note/project/media/;  
    }
}


Gunicorn Configuration:


gunicorn --workers 3 --bind 127.0.0.1:8000 gestion_notes_site.wsgi:application
